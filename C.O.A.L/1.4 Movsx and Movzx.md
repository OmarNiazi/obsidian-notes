```cpp
.386
.model flat, stdcall
.stack 4096

ExitProcess PROTO, dwExitCode: DWORD

.data
	ary1 db 1, -1, 2, -2

.code
main PROC
	movzx cx, ary1[0]	; cx = 00000000 00000001
	movsx dx, ary1[1]	; dx = 11111111 11111111 (sign-extended) which is -1 in 2s complement

	; if we were to do:
	movzx bx, ary1[1] ; bx = 000000000 11111111 (zero-extended) which is not -1 in 2s complement
	movsx bx, ary1[0] ; bx = 000000000 00000001 (sign-extended) which is 1

	; movsx eax, 0Fh ; Error, immediate vals are not accepted
	; movzx eax, 0Fh ; Error, immediate vals are not accepted
	; so we need to load the immediate into a register first
	mov al, 0Fh
	movsx bx, al ; bx = 00000000 00001111 (sign-extended) which is 15

	mov eax, offset ary1
	; movsx ebx, byte ptr [eax+3] ; Error, size must be specified for memory operand
	movsx ebx, byte ptr [eax+3] ; ebx = 11111111 11111110 (sign-extended) which is -2 in 2s complement

	invoke ExitProcess, 0
	
main ENDP
END main
```
